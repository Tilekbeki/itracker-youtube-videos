Зависит от контекста.
```HTML
<!-- Обычная кнопка -->
<button>Button</button>

<!-- Кнопка для отправки формы -->
<button type="submit">Button</button>
<input type="submit" value="button">

<!-- Кнопка инпут -->
<input type="submit">

<!-- Ссылочная кнопка -->
<a href='#'>Button</a>
```
